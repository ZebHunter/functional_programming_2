module type Set = sig
  type elt
  type t

  val empty : t
  val insert : elt -> t -> t
  val find : elt -> t -> elt option
  val delete : elt -> t -> t
  val map : (elt -> elt) -> t -> t
  val fold_left : ('b -> elt -> 'b) -> 'b -> t -> 'b
  val fold_right : (elt -> 'b -> 'b) -> t -> 'b -> 'b
  val filter : (elt -> bool) -> t -> t
  val concate : t -> t -> t
  val to_list : t -> elt list
end
